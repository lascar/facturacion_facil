🚀 Setup del Entorno de Desarrollo Avanzado

⚡ Configuración Completa para Desarrolladores

# 1. Clonar y configurar el entorno
git clone https://github.com/tu-repo/facturacion-facil.git
cd facturacion-facil

# 2. Crear entorno virtual con Python 3.8+
python3.8 -m venv venv_dev
source venv_dev/bin/activate  # Linux/Mac
# o
venv_dev\Scripts\activate     # Windows

# 3. Instalar dependencias de desarrollo
pip install -r requirements.txt
pip install -r requirements-dev.txt  # Testing, linting, etc.

# 4. Configurar herramientas de desarrollo
pre-commit install  # Git hooks
mypy --install-types  # Type checking

# 5. Ejecutar suite completa de tests
python -m pytest test/ -v --cov=. --cov-report=html

# 6. Análisis de código
flake8 .  # Linting
mypy .    # Type checking
bandit -r .  # Security analysis

# 7. Profiling de performance
python -m cProfile -o profile.stats main.py
python -c "import pstats; pstats.Stats('profile.stats').sort_stats('cumulative').print_stats(20)"

🔧 Comandos de Desarrollo

# Testing avanzado
python -m pytest test/unit/ -v          # Solo unit tests
python -m pytest test/integration/ -v   # Solo integration tests
python -m pytest -m "not slow" -v       # Excluir tests lentos
python -m pytest --lf                   # Solo tests que fallaron

# Análisis de calidad
flake8 --statistics                      # Linting con estadísticas
mypy --strict                           # Type checking estricto
bandit -r . -f json                     # Security scan en JSON
radon cc . --average                    # Complejidad ciclomática

# Profiling y performance
python -m cProfile -s cumulative main.py
python -m memory_profiler main.py
python -m line_profiler main.py

# Documentación
sphinx-build -b html docs/ docs/_build/
pdoc --html --output-dir docs/ .

# Deployment
python setup.py sdist bdist_wheel
twine check dist/*
pyinstaller --onefile --windowed main.py
